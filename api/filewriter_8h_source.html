73bool kinc_file_writer_open(kinc_file_writer_t *writer, const char *filepath) {
74 writer->file = NULL;
75 writer->mounted = false;
76#ifdef MOUNT_SAVES
77 if (!mountSaveData(true)) {
78 return false;
79 }
80 writer->mounted = true;
81#endif
82 char path[1001];
83 strcpy(path, kinc_internal_save_path());
84 strcat(path, filepath);
85
86#ifdef KINC_WINDOWS
87 wchar_t wpath[MAX_PATH];
88 MultiByteToWideChar(CP_UTF8, 0, path, -1, wpath, MAX_PATH);
89 writer->file = CreateFileW(wpath, GENERIC_WRITE, 0, NULL, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
90#else
91 writer->file = fopen(path, "wb");
92#endif
93 if (writer->file == NULL) {
94 kinc_log(KINC_LOG_LEVEL_WARNING, "Could not open file %s.", filepath);
95 return false;
96 }
97 return true;
98}
