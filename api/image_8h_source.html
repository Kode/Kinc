183 static void *buffer_realloc(void *p, size_t size) {
184  if (p == NULL) {
185  return buffer_malloc(size);
186  }
187  uint8_t *old_pointer = (uint8_t *)p;
188  size_t old_size = *(size_t *)(old_pointer - sizeof(size_t));
189  if (size <= old_size) {
190  return old_pointer;
191  }
192  else {
193  if (last_allocated_pointer == old_pointer) {
194  size_t last_size = &buffer[buffer_offset] - old_pointer;
195  size_t size_diff = size - last_size;
196  buffer_offset += size_diff + sizeof(size_t);
197  return old_pointer;
198  }
199  uint8_t *new_pointer = (uint8_t *)buffer_malloc(size);
200  if (new_pointer == NULL) {
201  return NULL;
202  }
203  memcpy(new_pointer, old_pointer, old_size < size ? old_size : size);
204  return new_pointer;
205  }
206 }
