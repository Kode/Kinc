217#define BUFFER_SIZE 4096 * 4096 * 4
218static uint8_t buffer[BUFFER_SIZE];
219static size_t buffer_offset = 0;
220static uint8_t *last_allocated_pointer = 0;
221
222static void *buffer_malloc(size_t size) {
223 uint8_t *current = &buffer[buffer_offset];
224 buffer_offset += size + sizeof(size_t);
225 if (buffer_offset > BUFFER_SIZE) {
226 kinc_log(KINC_LOG_LEVEL_ERROR, "Not enough memory on image.c Buffer.");
227 return NULL;
228 }
229 memcpy(current, &size, sizeof(size_t));
230 last_allocated_pointer = current + sizeof(size_t);
231 return current + sizeof(size_t);
232}
233
234static void *buffer_realloc(void *p, size_t size) {
235 if (p == NULL) {
236 return buffer_malloc(size);
237 }
238 uint8_t *old_pointer = (uint8_t *)p;
239 size_t old_size = *(size_t *)(old_pointer - sizeof(size_t));
240 if (size <= old_size) {
241 return old_pointer;
242 }
243 else {
244 if (last_allocated_pointer == old_pointer) {
245 size_t last_size = &buffer[buffer_offset] - old_pointer;
246 size_t size_diff = size - last_size;
247 buffer_offset += size_diff + sizeof(size_t);
248 return old_pointer;
249 }
250 uint8_t *new_pointer = (uint8_t *)buffer_malloc(size);
251 if (new_pointer == NULL) {
252 return NULL;
253 }
254 memcpy(new_pointer, old_pointer, old_size < size ? old_size : size);
255 return new_pointer;
256 }
257}

301static _Bool endsWith(const char *str, const char *suffix) {
302 if (str == NULL || suffix == NULL)
303 return 0;
304 size_t lenstr = strlen(str);
305 size_t lensuffix = strlen(suffix);
306 if (lensuffix > lenstr)
