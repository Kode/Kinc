38static void readChunk(uint8_t **data, struct WaveData *wave) {
39 char fourcc[5];
40 readFOURCC(data, fourcc);
41 uint32_t chunksize = kinc_read_u32le(*data);
42 *data += 4;
43 if (strcmp(fourcc, "fmt ") == 0) {
44 wave->audioFormat = kinc_read_u16le(*data + 0);
45 wave->numChannels = kinc_read_u16le(*data + 2);
46 wave->sampleRate = kinc_read_u32le(*data + 4);
47 wave->bytesPerSecond = kinc_read_u32le(*data + 8);
48 wave->bitsPerSample = kinc_read_u16le(*data + 14);
49 *data += chunksize;
50 }
51 else if (strcmp(fourcc, "data") == 0) {
52 wave->dataSize = chunksize;
53 wave->data = (uint8_t *)malloc(chunksize * sizeof(uint8_t));
54 kinc_affirm(wave->data != NULL);
55 memcpy(wave->data, *data, chunksize);
56 *data += chunksize;
57 }
58 else {
59 *data += chunksize;
60 }
61}

107kinc_a1_sound_t *kinc_a1_sound_create_from_buffer(uint8_t *audio_data, const uint32_t size, kinc_a1_audioformat_t format) {
108 kinc_a1_sound_t *sound = find_sound();
109 assert(sound != NULL);
110 sound->in_use = true;
111 sound->volume = 1.0f;
112 sound->size = 0;
113 sound->left = NULL;
114 sound->right = NULL;
115 // size_t filenameLength = strlen(filename);
116 uint8_t *data = NULL;
117
118 if (format == KINC_A1_AUDIOFORMAT_OGG) {
119 int channels, sample_rate;
120 int samples = stb_vorbis_decode_memory(audio_data, size, &channels, &sample_rate, (short **)&data);
121 kinc_affirm(samples > 0);
122 sound->channel_count = (uint8_t)channels;
123 sound->samples_per_second = (uint32_t)sample_rate;
124 sound->size = samples * 2 * sound->channel_count;
125 sound->bits_per_sample = 16;
126 }
127 else if (format == KINC_A1_AUDIOFORMAT_WAV) {
128 struct WaveData wave = {0};
129 {
130 uint8_t *data = audio_data;
131
132 checkFOURCC(&data, "RIFF");
133 uint32_t filesize = kinc_read_u32le(data);
134 data += 4;
135 checkFOURCC(&data, "WAVE");
136 while (data + 8 - audio_data < (intptr_t)filesize) {
137 readChunk(&data, &wave);
138 }
139 }
140
141 sound->bits_per_sample = (uint8_t)wave.bitsPerSample;
142 sound->channel_count = (uint8_t)wave.numChannels;
143 sound->samples_per_second = wave.sampleRate;
144 data = wave.data;
145 sound->size = wave.dataSize;
146 }
147 else {
148 kinc_affirm(false);
149 }
150
151 if (sound->channel_count == 1) {
152 if (sound->bits_per_sample == 8) {
153 sound->left = (int16_t *)malloc(sound->size * sizeof(int16_t));
154 sound->right = (int16_t *)malloc(sound->size * sizeof(int16_t));
155 splitMono8(data, sound->size, sound->left, sound->right);
156 }
157 else if (sound->bits_per_sample == 16) {
158 sound->size /= 2;
159 sound->left = (int16_t *)malloc(sound->size * sizeof(int16_t));
160 sound->right = (int16_t *)malloc(sound->size * sizeof(int16_t));
161 splitMono16((int16_t *)data, sound->size, sound->left, sound->right);
162 }
163 else {
164 kinc_affirm(false);
165 }
166 }
167 else {
168 // Left and right channel are in s16 audio stream, alternating.
169 if (sound->bits_per_sample == 8) {
170 sound->size /= 2;
171 sound->left = (int16_t *)malloc(sound->size * sizeof(int16_t));
172 sound->right = (int16_t *)malloc(sound->size * sizeof(int16_t));
173 splitStereo8(data, sound->size, sound->left, sound->right);
174 }
175 else if (sound->bits_per_sample == 16) {
176 sound->size /= 4;
177 sound->left = (int16_t *)malloc(sound->size * sizeof(int16_t));
178 sound->right = (int16_t *)malloc(sound->size * sizeof(int16_t));
179 splitStereo16((int16_t *)data, sound->size, sound->left, sound->right);
180 }
181 else {
182 kinc_affirm(false);
183 }
184 }
185 sound->sample_rate_pos = kinc_a2_samples_per_second() / (float)sound->samples_per_second;
186 free(data);
187
188 return sound;
189}

191kinc_a1_sound_t *kinc_a1_sound_create(const char *filename) {
192 size_t filenameLength = strlen(filename);
193
194 kinc_a1_audioformat_t fileformat;
195 if (strncmp(&filename[filenameLength - 4], ".ogg", 4) == 0) {
196 fileformat = KINC_A1_AUDIOFORMAT_OGG;
197 }
198 else if (strncmp(&filename[filenameLength - 4], ".wav", 4) == 0) {
199 fileformat = KINC_A1_AUDIOFORMAT_WAV;
200 }
201 else {
202 assert(false);
203 }
204
205 kinc_file_reader_t file;
206 if (!kinc_file_reader_open(&file, filename, KINC_FILE_TYPE_ASSET)) {
207 return NULL;
208 }
209
210 uint8_t *filedata = (uint8_t *)malloc(kinc_file_reader_size(&file));
211 kinc_file_reader_read(&file, filedata, kinc_file_reader_size(&file));
212 kinc_file_reader_close(&file);
213 size_t filesize = kinc_file_reader_size(&file);
214
215 kinc_a1_sound_t *sound = kinc_a1_sound_create_from_buffer(filedata, (uint32_t)filesize, fileformat);
216
217 free(filedata);
218
219 return sound;
220}
