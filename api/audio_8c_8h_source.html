21static float sampleLinear(int16_t *data, double position) {
22 int pos1 = (int)position;
23 int pos2 = (int)(position + 1);
24 float sample1 = data[pos1] / 32767.0f;
25 float sample2 = data[pos2] / 32767.0f;
26 float a = (float)(position - pos1);
27 return sample1 * (1 - a) + sample2 * a;
28}

150kinc_a1_channel_t *kinc_a1_play_sound(kinc_a1_sound_t *sound, bool loop, float pitch, bool unique) {
151 kinc_a1_channel_t *channel = NULL;
152 kinc_mutex_lock(&mutex);
153 bool found = false;
154 for (int i = 0; i < CHANNEL_COUNT; ++i) {
155 if (channels[i].sound == sound) {
156 found = true;

158 }
159 }
160 if (!found || !unique) {
161 for (int i = 0; i < CHANNEL_COUNT; ++i) {
162 if (channels[i].sound == NULL) {
163 channels[i].sound = sound;
164 channels[i].position = 0;
165 channels[i].loop = loop;
166 channels[i].pitch = pitch;
167 channels[i].volume = 1.0f;
168 channel = &channels[i];
169 break;
170 }
171 }
172 }
173 kinc_mutex_unlock(&mutex);
174 return channel;
