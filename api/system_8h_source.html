677void kinc_load_save_file(const char *filename) {
678 free(current_file);
679 current_file = NULL;
680 current_file_size = 0;
681
682 kinc_file_reader_t reader;
683 if (kinc_file_reader_open(&reader, filename, KINC_FILE_TYPE_SAVE)) {
684 current_file_size = kinc_file_reader_size(&reader);
685 current_file = (uint8_t *)malloc(current_file_size);
686 kinc_file_reader_read(&reader, current_file, current_file_size);
687 kinc_file_reader_close(&reader);
688 }
689}
