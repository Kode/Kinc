242void kinc_internal_mouse_trigger_move(int window, int x, int y) {
243 int movementX = 0;
244 int movementY = 0;
245 if (kinc_mouse_is_locked()) {
246 movementX = x - centerX;
247 movementY = y - centerY;
248 if (movementX != 0 || movementY != 0) {
249 kinc_mouse_set_position(window, centerX, centerY);
250 x = centerX;
251 y = centerY;
252 }
253 }
254 else if (moved) {
255 movementX = x - lastX;
256 movementY = y - lastY;
257 }
258 moved = true;
259
260 lastX = x;
261 lastY = y;
262 if (mouse_move_callback != NULL && (movementX != 0 || movementY != 0)) {
263 mouse_move_callback(window, x, y, movementX, movementY, mouse_move_callback_data);
264 }
265}

271void kinc_mouse_lock(int window) {
272 if (!kinc_mouse_can_lock()) {
273 return;
274 }
275 locked = true;
276 kinc_internal_mouse_lock(window);
277 kinc_mouse_get_position(window, &preLockX, &preLockY);
278 centerX = kinc_window_width(window) / 2;
279 centerY = kinc_window_height(window) / 2;
280 kinc_mouse_set_position(window, centerX, centerY);
281}
